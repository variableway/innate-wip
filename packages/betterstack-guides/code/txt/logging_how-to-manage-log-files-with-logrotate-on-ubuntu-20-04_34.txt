/home/<user>/logify/*.log
{
    hourly
    missingok
    rotate 7
    compress
    notifempty
    [highlight]
    sharedscripts
    postrotate
      curl https://betteruptime.com/api/v1/heartbeat/<heartbeat_id>
    endscript
    [/highlight]
}
