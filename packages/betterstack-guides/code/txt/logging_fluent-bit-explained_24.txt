[label fluent-bit-config/fluent-bit.conf]
[SERVICE]
    Flush        1
    Daemon       off
    Log_Level    debug
    Parsers_File parser_json.conf

[INPUT]
    Name         tail
    Path         /var/log/logify/app.log
    Parser       json_parser
    Tag          filelogs

[highlight]
[FILTER]
    Name record_modifier
    Match filelogs
    Remove_key     emailAddress
    Record hostname ${HOSTNAME}
[/highlight]

[OUTPUT]
    Name         stdout
    format       json
    Match        filelogs
