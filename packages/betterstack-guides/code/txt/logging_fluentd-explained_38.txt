<filter file.logs>
  @type record_transformer
  enable_ruby true
  remove_keys emailAddress
  <record>
    hostname "#{Socket.gethostname}"
    timestamp ${Time.at(record["timestamp"]).strftime("%Y-%m-%dT%H:%M:%S.%L%z")}
    [highlight]
    is_successful ${record["status"] == 200 ? "true" : "false"}
    [/highlight]
  </record>
</filter>
