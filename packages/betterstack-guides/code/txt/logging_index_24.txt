#!/bin/bash

tail -f /var/log/auth.log | grep --line-buffered "Failed password" | while read line
do
  echo "$(date): SSH authentication failure detected" | mail -s "SSH Alert: Failed Login" admin@example.com
  echo "$line" >> /var/log/ssh_alerts.log
done
