[label logstash/config/logstash.conf]
input {
  file {
    path => "/var/log/logify/app.log"
    start_position => "beginning"
  }
}
[highlight]
filter {
  if [message] =~ /^{.*}$/ {
    json {
      source => "message"
    }
  }
}
[/highlight]

  stdout {
    codec => rubydebug
  }
}
